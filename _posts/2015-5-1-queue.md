## queue
queue（队列）也是常见的数据结构。先进先出的是其显著特点.主要操作
- enqueue(入队)
- dequeue(出队)


## 代码实现与分析

```
//queue.c
#include <stdio.h>
#include <stdlib.h>

#if 0
1 这里是队列的简单链表实现
2 head 是一个分配了空间的固定节点 但是不用来保存element
3 tail 是一个指针 而且一直不断的变化

#endif

typedef int Element;

typedef struct mynode {
    Element element;
    struct mynode* next;
}node;

typedef struct myqueue {
    node* head;
    node* tail;
}queue;

typedef queue* QUEUE;

static QUEUE init_queue(void);
static void clear_queue(QUEUE);
static void destroy_queue(QUEUE);

static void enqueue(QUEUE, Element);
static Element dequeue(QUEUE);

static node* create_node(Element );
static int is_null(QUEUE);
static int is_exist(QUEUE);
static void print_queue(QUEUE);

QUEUE init_queue(void)
{
    node* head = (node*)malloc(sizeof(node));
    head->element = 0;
    head->next = NULL;

    QUEUE queue = (QUEUE)malloc(sizeof(queue));
    queue->tail=NULL;
    queue->head = head;

    return queue;

}
void clear_queue(QUEUE queue)
{
    node* head = queue->head;
    node* p = head->next;
    node* pnext ;
    while(p)
    {
        pnext = p->next;
        free(p);
        p= pnext;
    }
}

void destroy_queue(QUEUE queue)
{
    clear_queue(queue);
    free(queue->head);
    free(queue);
}


void enqueue(QUEUE queue, Element e)
{
    node* newnode = create_node(e);
    node* head= queue->head;
    node* tail = queue->tail;
    if(tail==NULL)
    {
        head->next = newnode;
    }
    else
    {
        tail->next = newnode;
    }
    queue->tail= newnode;
    printf("enqueue: %d \n",e);
}
//调用函数前需要判断队列是否为空
Element dequeue(QUEUE queue)
{
    node* head = queue->head;
    node* p = head->next;
    Element e = p->element;
    node* pnext = p->next;
    free(p);
    head->next = pnext;
    if(head->next==NULL)
    {//空队列
        queue->tail=NULL;
    }
    printf("dequeue: %d \n",e);
    return e;
}

node* create_node(Element e)
{
    node* p = (node*)malloc(sizeof(node));
    p->element = e;
    p->next =NULL;
    return p;
}

int is_null(QUEUE queue)
{
    if(queue->head->next==NULL)return 1;
    return 0;
}
int is_exist(QUEUE queue)
{
    if(queue)return 1;
    return 0;
}
void print_queue(QUEUE queue)
{
    if(!is_exist(queue))return;
    if(is_null(queue))return;
    node* p = queue->head->next;
    while(p)
    {
        printf("element: %d \n",p->element);
        p = p->next;
    }
}

int main()
{
    QUEUE qu = init_queue();
    Element e;

    enqueue(qu,1);
    enqueue(qu,3);
    enqueue(qu,5);
    enqueue(qu,7);
    printf("11111\n");

    print_queue(qu);
    if(!is_null(qu)) e = dequeue(qu);
 
    
    enqueue(qu,9);
    if(!is_null(qu)) e = dequeue(qu);
    if(!is_null(qu)) e = dequeue(qu);
    printf("222222\n");

    print_queue(qu);

    destroy_queue(qu);



    return 0;
}
```
