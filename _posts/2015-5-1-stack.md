## stack
stack（栈）是一种基本的数据结构。
主要包括三种操作
- push	压栈（添加数据） 
- pop	弹出数据（删除）
- top	查看


## 实现代码与分析 

```
//STACK*.c
#include <stdio.h>
#include <stdlib.h>

#if 0
1 这里是一个栈的简单链表实现
2 这里的头结点也就是head 也是一个节点 分配了空间 只是没有有效元素数据 
3 这里的尾节点也就是tail 只是一个指针 没有分配对应的空间
3 is_null 是判断列表中有效数据节点 head不算有效数据节点

typedef int Element;

typedef struct mynode {
    Element element;
    struct mynode* next;
}node;


typedef struct mystack {
    node* head;
    node* tail;
}STACK;

static STACK* init_stack(void);
static void clear_stack(STACK*);
static void destroy_stack(STACK*);

static Element top(STACK*);
static void push(STACK*, Element);
static Element pop(STACK*);
static node* create_node(Element );
static int is_null(STACK*);
static int is_exist(STACK*);

static void print_stack(STACK*);

STACK* init_stack(void)
{
    node* head = (node*)malloc(sizeof(node));
    head->element = 0;
    head->next = NULL;
    node* tail = NULL;

    STACK* p = (STACK*)malloc(sizeof(STACK));
    p->head = head;
    p->tail = tail;

    return p;
}

void clear_stack(STACK* stack)
{
    if(!is_exist(stack))return;
    if(is_null(stack))return;
    node* head = stack->head;
    node* tail = stack->tail;
    node* p = head->next;
    node* pnext ;
    while(p)
    {
        pnext = p->next;
        free(p);
        p = pnext;
    }
    head->next = NULL;
    tail= NULL;

}
void destroy_stack(STACK* stack)
{
    if(!is_exist(stack))return;
    clear_stack(stack);
    free(stack->head);
    free(stack);
}

int is_exist(STACK* stack)
{
    if(stack==NULL)
    {
        printf("no stack !!!\n");
        return 0;
    }
    return 1;
}
int is_null(STACK* stack)
{
    if(stack->head->next==NULL)return 1;
    return 0;
}
node* create_node(Element e)
{
    node* p = (node*)malloc(sizeof(node));
    p->element = e;
    p->next = NULL;
    return p;
}

//调用前请先判断栈是否为空
Element top(STACK* stack)
{
     return stack->tail->element;
}

void push(STACK* stack, Element e)
{
    node* p = create_node(e);
    if(is_null(stack))
    {//插入的节点直接在head后面
        stack->head->next = p;
    }
    else
    {//插入的节点在tail后面
        stack->tail->next = p;

    }   
    stack->tail = p;
}
//调用前请先判断栈是否为空 
//注意只有一个节点的情况
Element pop(STACK* stack)
{
    Element e =  stack->tail->element;
    printf("pop %d\n",e);
    node* p = stack->head->next;//获取当前处理的节点
    node* pbefore=stack->head;//保留当前处理的节点的前置节点
    while(p!=stack->tail)
    {
        pbefore = p;
        p = p->next;
    }
    free(p);
    pbefore->next = NULL;//新的tail产生
    stack->tail = pbefore;
    //是否为空栈
    if(stack->tail==stack->head)
    {
        stack->tail= NULL;
    }
    return e;
}

void print_stack(STACK* stack)
{
    if(!is_exist(stack))return;
    if(is_null(stack))return;
    node* head = stack->head;
    node* p = head->next;
    while(p)
    {
        printf("element: %d\n",p->element);
        p = p->next;
    }

}
int main()
{
    STACK * stack = init_stack();
    push(stack,1);
    push(stack,3);
    push(stack,5);
    push(stack,7);

    printf("pirnt1 \n");
    print_stack(stack);

    push(stack,9);
    push(stack,11);
    //pop前要判断是否为空
    pop(stack);
    pop(stack);
    printf("pirnt2 \n");
    print_stack(stack);

    destroy_stack(stack);
    return 0;
}
```
